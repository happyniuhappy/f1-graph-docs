      # OLContextMenu 右键菜单

      # 地图上右键菜单， 支持多种方法添加

      <template>
<div style="height: 400px; width: 100%">
    <!-- 地图组件 -->
    <OLMap>
        <!-- 基础底图高德瓦片 -->
        <AMapTileLayer />
        <OLVectorLayer title="矢量图层1">
            <!-- 矢量数据源组件 -->
            <OLVectorSource ref="targetSource">
            </OLVectorSource>
        </OLVectorLayer>
        <!-- 绘制功能 -->
        <OLBar :toggleOne="true">  
            <OLToggle :active="false" :title="'画点'" icon="fa fa-circle" @onToggle="_onToggle($event, 'Point')">
                <OLDraw ref="drawPoint" geoType="Point"/>
            </OLToggle>
        </OLBar>
        <OLContextMenu :menus="contextMenu" />
    </OLMap>
  </div>
</template>
<script>
export default {
    data: function(){
        return {
            contextMenu: [
                {
                    id: "delete",
                    text: "删除",
                    onClick: function(evt, fea, layer) {
                    console.log("点击删除图元");
                    console.log(fea);
                    /**
                     *  删除方法
                     *  layer.getSource().removeFeature(fea);
                     */
                    }
                },
                {
                    id: "parameter",
                    text: "台账",
                    onClick: function(evt, fea, layer) {
                        console.log(fea)
                        /**
                        *  台账 
                        *  fea.getxxxx   方法获取当前图元属性
                        */
                    }
                }
            ]
        }
    },

    methods: {
        _onToggle: function(active, type) {
            if(active){
                let source = this.$refs.targetSource.mapObject;
                this.$refs['draw'+type].setSource(source);
            }
        }
    }
}
</script>

      # 示例

      {  
    id: "parameter",   
    text: "台账",  
    onClick: function(evt, fea, layer) {  
        console.log(fea)  
    }   
}

id： 菜单Id值  
text： 菜单显示文字（最多8个中文字符）  
onClick：function(evt, fea, layer){} — 菜单点击事件  
evt —— 点击事件event  
fea —— 当右键点击图元时，返回点击图元Feature数据，若点击空白返回undefined  
layer —— 返回右键单击图元所在Layer
