      # OLDistanceSearch范围搜索

      范围查询功能是以地图上某一位置为中心，查找指定距离内的兴趣点。

      # 通过点击地图方式选取中心点

      点击按钮，激活范围搜索功能，点击地图选取中心点，默认搜素1000m内的图形。

      <template>
    <div style="height: 400px; width: 100%">
        <!-- 地图组件 -->
        <OLMap>
            <!-- 基础底图高德瓦片 -->
            <AMapTileLayer />
            <!-- 矢量图层组件 -->
            <OLVectorLayer title="矢量图层1">
                <!-- 矢量数据源组件 -->
                <OLVectorSource ref="targetSource">
                    <!-- 图形元素 -->
                    <OLFeature v-for="(f, i) in features" :key="i" :geoType="f.geometry.type" :coordinates="f.geometry.coordinates" :properties="f.properties"></OLFeature>
                </OLVectorSource>
            </OLVectorLayer>
            <!-- 范围查询 -->
            <OLBar :toggleOne="true" :position="pos">
                <OLToggle :title="title" :icon="icon" @onToggle="_onToggleSource">
                    <OLDistanceSearch @mapclick="onMapClick" ref="distanceSearch"/>
                </OLToggle>
            </OLBar>
        </OLMap>
    </div>
</template>
<script>
export default {
    data: function(){
        return {
            title: '以地图点击位置为中心搜素1000m内的图形要素',
            icon: 'fa fa-dot-circle-o',
            features: [
                
                {
                    "type": "Feature",
                    "properties": {
                        "name": "中恒博瑞"
                    },
                    "geometry": {
                        "type": "Point",
                        "coordinates": [116.302778,40.050833]
                    }
                },
                {
                  "type": "Feature",
                    "properties": {
                        "name": "TEST1"
                    },
                    "geometry": {
                        "type": "Point",
                        "coordinates": [116.311402,40.057638]
                    }
                },
                {
                  "type": "Feature",
                    "properties": {
                        "name": "TEST2"
                    },
                    "geometry": {
                        "type": "Point",
                        "coordinates": [116.309201,40.056734]
                    }
                },
                {
                  "type": "Feature",
                    "properties": {
                        "name": "TEST3"
                    },
                    "geometry": {
                        "type": "Point",
                        "coordinates": [116.313091,40.055892]
                    }
                }
            ],
            // 工具栏位置
            pos: 'top'
        }
    },

    methods: {

        /**
         * 激活功能按钮后，设置搜索目标数据源
         * @private
         */
        _onToggleSource: function(active) {
            if(active){
                let source = this.$refs.targetSource.mapObject;
                this.$refs.distanceSearch.setSource(source);
            }
        },

        onMapClick: function(evt) {
            console.log(evt);
            this.$alert('在范围内存在搜索点个数为：' + evt.sreachFeatures.length);
        }
    }
}
</script>

      # 属性

      参数 说明 类型 可选值 默认值
      active 是否激活范围搜索功能 Boolean true/false false
      displayInLayerSwitcher 是否添加临时层 Boolean true/false false
      radius 查询半径，默认1000m Number — 1000
      center 设置查询中心点 Array — [0, 0]

      # 事件

      事件名称 说明 回调参数
      setSource 设置搜索源 需要设置查询的数据源
