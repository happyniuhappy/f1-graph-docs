      # Popup弹窗

      <template>
    <!-- 外层容器 -->
    <div style="height: 400px; width: 100%">
        <!-- 地图组件，注册鼠标移动事件 -->
        <OLMap :zoom="false" :center="initCenter" @pointermove="_onPointermove">
            <!-- 基础底图高德瓦片 -->
            <AMapTileLayer></AMapTileLayer>
            <!-- 信息弹框 -->
            <OLPopup v-if="popupVisible" :coordinate="popupCoordinates" :closeBox="isCloseBox" positioning="center-left" popupClass="tips shadow">
                <!-- 信息弹框内容，自定义html代码 -->
                <div class="jbgraph-popup">id：{{properties.id}}</div>
                <div class="jbgraph-popup">名称：{{properties.name}}</div>
            </OLPopup>
            <!-- 设备数据图层 -->
            <OLVectorLayer>
                <!-- 设备数据源 -->
                <OLGeoJsonSource :features="features"></OLGeoJsonSource>
            </OLVectorLayer>
        </OLMap>
    </div>
</template>

<script>
export default {

    data: function() {
        return {
            // 地图初始化位置
            initCenter: [103.778611, 36.066111],
            // 信息框是否可见
            popupVisible: false,
            // 信息框是否显示关闭按钮
            isCloseBox: false,
            // 信息框坐标
            popupCoordinates: [],
            // 信息框属性
            properties: {},
            // 图形要素集合对象
            features: [
                {
                    type: "Feature",
                    properties: {
                        id: "id_1",
                        name: "金雨大厦",
                        style: {
                            visible: true,
                            image: {
                                circle: {
                                    fill: "green",
                                    stroke: { color: "blue", width: 2 },
                                    radius: 10
                                }
                            }
                        }
                    },
                    geometry: {
                        type: "Point",
                        coordinates: [103.778611, 36.066111]
                    }
                },
                {
                    type: "Feature",
                    properties: {
                        id: "id_2",
                        name: "110kV七里河变电站-导出坐标"
                    },
                    geometry: {
                        type: "Point",
                        coordinates: [103.770654454588, 36.0838336262833]
                    }
                },
                {
                    type: "Feature",
                    properties: {
                        name: "110kV七里河变电站-拾取坐标"
                    },
                    geometry: {
                        type: "Point",
                        coordinates: [103.76904353, 36.08082923]
                    }
                },
                {
                    type: "Feature",
                    properties: {
                        name: "中恒博瑞"
                    },
                    geometry: {
                        type: "Point",
                        coordinates: [116.302778, 40.050833]
                    }
                },
                {
                    type: "Feature",
                    properties: {
                        id: "id_1-id_2",
                        name: "金雨大厦-110kV七里河变电站(导出坐标)",
                        source: { id: "id_1" },
                        target: { id: "id_2" },
                        style: {
                            stroke: { color: "green", width: 2 }
                        }
                    },
                    geometry: {
                        type: "LineString",
                        // "coordinates": [[116.302778,40.050833], [103.770654454588,36.0838336262833]]
                        "coordinates": [[103.778611,36.066111], [103.770654454588,36.0838336262833]]
                    }
                }
            ]
        };
    },

    methods: {

        /**
         * 地图鼠标移动事件执行方法
         * @private
         * @param {Event} 事件参数
         * @return {void}
         */
        _onPointermove: function(evt) {
            // 根据鼠标当前移动的位置查找图形元素
            var feature = evt.map.forEachFeatureAtPixel(evt.pixel, feature => feature);
            // 如果查找到图形元素，并且是点图形，显示信息框
            if (feature && feature.getGeometry().getType() == 'Point') {
                // 显示信息弹框
                this.popupVisible = true;
                // 从features要素集合中筛选出对应的对象
                // 这里用name属性进行匹配，具体业务具体分析
                let fs = this.features.filter(f => {
                    return feature.get("name")===f.properties.name;
                });
                // 设置弹框的锚点位置坐标，格式是经纬度坐标（无偏移）
                this.popupCoordinates = fs[0].geometry.coordinates;
                // 设置弹框显示的内容属性
                this.properties = {
                    id: feature.get("id"),
                    name: feature.get("name")
                };
            } else {
                // 隐藏信息弹框
                this.popupVisible = false;
            }
        }
    }
};
</script>
