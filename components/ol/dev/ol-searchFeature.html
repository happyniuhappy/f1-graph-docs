      # 默认使用

      地图自动将设置搜索到的图形为地图中心

      <template>
<div style="height: 400px; width: 100%">
    <!-- 地图组件 -->
    <OLMap>
        <!-- 基础底图高德瓦片 -->
        <AMapTileLayer />
        <!-- 矢量图层组件 -->
        <OLVectorLayer title="矢量图层1">
            <!-- 矢量数据源组件 -->
            <OLVectorSource ref="targetSource">
                <!-- 图形元素 -->
                <OLFeature :geoType="geoType" :coordinates="coordinates" :properties="properties"></OLFeature>
            </OLVectorSource>
        </OLVectorLayer>
        <OLSearchFeature/>
    </OLMap>
  </div>
</template>

<script>
export default {
    data: function(){
        return {
            // 几何类型
            geoType: "Point",
            // 经纬度坐标
            coordinates: [116.302778,40.050833],
            // 属性信息
            properties: {
                name: "中恒博瑞",
            }
        }
    }
}
</script>

      # 搜素完成进行地图缩放操作

      搜索并进行地图缩放

      <template>
<div style="height: 400px; width: 100%">
    <!-- 地图组件 -->
    <OLMap ref="mapComp">
        <!-- 基础底图高德瓦片 -->
        <AMapTileLayer />
        <!-- 矢量图层组件 -->
        <OLVectorLayer title="矢量图层1">
            <!-- 矢量数据源组件 -->
            <OLVectorSource ref="targetSource">
                <!-- 图形元素 -->
                <OLFeature :geoType="geoType" :coordinates="coordinates" :properties="properties"></OLFeature>
            </OLVectorSource>
        </OLVectorLayer>
        <OLSearchFeature @ready="_onReady" ref="search"/>
    </OLMap>
  </div>
</template>

<script>
export default {
    data: function(){
        return {
            // 几何类型
            geoType: "Point",
            // 经纬度坐标
            coordinates: [116.302778,40.050833],
            // 属性信息
            properties: {
                name: "中恒博瑞",
            }
        }
    },

    methods: {
        _onReady(mapObject) {
            let me = this;
            mapObject.on("select", mapEvent => {
                /* me.$refs.mapComp.moveCenter({
                    center: me.coordinates,
                    zoom: 13
                }); */
                me.$refs.mapComp.setZoomLevel(13);
            });
        }
    }
}
</script>
