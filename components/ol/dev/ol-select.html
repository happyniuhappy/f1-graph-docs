      选中矢量图层交互操作组件

      # 单选

      # 选中图形

      鼠标移动到要选择的图形之上，单击鼠标左键选中图形，图形变为深蓝色显示（默认显示），此时图形处于被选中状态。

      # 取消选中

      单击空白处，取消图形选中，选中图形样式恢复到原来。

      # 框选

      # 选中图形

      按下ctrl键不要松开，鼠标移动到合适区域并按下鼠标左键不要松开，同时进行拉框操作（即移动鼠标绘制矩形），绘制完成松开鼠标与ctrl键，与矩形交叉的图形将被选中。

      # 取消选中

      单击空白处，取消图形选中，选中图形样式恢复到原来。

      直接嵌入到地图中，OLMap作为父容器。

            <!-- 基础底图，百度地图瓦片 -->
            <BMapTileLayer></BMapTileLayer>
            <!-- 矢量图层组件 -->
            <OLVectorLayer title="矢量图层1">
                <!-- 矢量数据源组件 -->
                <OLVectorSource>
                    <!-- 图形元素 -->
                    <OLFeature v-for="(f, i) in features" :key="i" :geoType="f.geometry.type" :coordinates="f.geometry.coordinates" :properties="f.properties"></OLFeature>
                </OLVectorSource>
            </OLVectorLayer>
            <!-- 图形选中组件，注册选中事件，默认交互功能不激活，这里需要激活组件功能 -->
            <OLSelect @select="_onSelectFeature" :active="isActive" ref="select"/>
        </OLMap>
    </div>
</template>

      # 嵌入按钮

      通过点击地图中状态按钮控制地图的图形选中功能

      嵌入到功能按钮中，通过按钮激活选中功能。

      <template>
    <!-- 外层容器 -->
    <div style="height: 400px; width: 100%">
        <!-- 地图组件 -->
        <OLMap :zoom="true">
            <!-- 基础底图，百度地图瓦片 -->
            <BMapTileLayer></BMapTileLayer>
            <!-- 矢量图层组件 -->
            <OLVectorLayer title="矢量图层1">
                <!-- 矢量数据源组件 -->
                <OLVectorSource>
                    <!-- 图形元素 -->
                    <OLFeature v-for="(f, i) in features" :key="i" :geoType="f.geometry.type" :coordinates="f.geometry.coordinates" :properties="f.properties"></OLFeature>
                </OLVectorSource>
            </OLVectorLayer>
            <!-- 根工具栏 -->
            <OLBar position="left-top">
                <!-- 状态切换按钮 -->
                <OLToggle icon="icon-gis-select-box" title="选中图形元素">
                    <!-- 图形选中组件，激活状态将与父容器按钮的状态保持一致，注册选中事件 -->
                    <OLSelect @select="_onSelectFeature" ref="select"/>
                    <!-- 子工具栏，其内部的按钮操作依赖于选中功能，只有激活选中图形功能才能进行子菜单内功能操作 -->
                    <OLBar>
                        <!-- 单击按钮删除选中图形 -->
                        <OLButton label="x" title="删除" @onClick="deleteFeatures"></OLButton>
                        <OLButton label="i" title="信息"></OLButton>
                    </OLBar>
                </OLToggle>
            </OLBar>
        </OLMap>
    </div>
</template>

        /**
         * 删除按钮单击事件执行方法
         * @private
         * @return {void}
         */
        deleteFeatures: function() {
            // 先获取选中组件的OL原生对象，在获取选中集合对象
            let selectFeatures = this.$refs.select.mapObject.getFeatures();
            // 删除选中的图形
            selectFeatures.forEach(f => {
                // 获取选中图形所在的图层
                let layer = this.$refs.select.mapObject.getLayer(f);
                // 如果图层存在，并获取数据源，最后从数据源中将图形删除
                layer && layer.getSource().removeFeature(f);
            });
            // 清除选中结果
            selectFeatures.clear();
        }
    }
}
