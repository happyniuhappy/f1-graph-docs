      # OLTextButton 文字按钮

      地图矢量图层中简单的按钮控件，可以编写html结构

      <template>
<div style="height: 400px; width: 100%">
    <!-- 地图组件 -->
    <OLMap>
        <!-- 基础底图高德瓦片 -->
        <AMapTileLayer />
        <OLBar :toggleOne="true" :position="'top-left'">
            <OLToggle :icon="'maki-marker_stroked'">
                <OLBar>
                    <OLTextButton :title="'test1'" :label="label1"/>
                    <OLTextButton :title="'test2'" :label="label2"/>
                </OLBar>
            </OLToggle>
        </OLBar>
    </OLMap>
  </div>
</template>

<script>
export default {
    data: function(){
        return {
            icon: "maki2-star",
            label1: "<div>Test1</div>",
            label2: "<div>Test2</div>"
        }
    }
}
</script>
