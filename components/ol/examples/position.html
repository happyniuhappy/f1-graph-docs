      # 地图定位

      # 定位方法

      OLMap.moveCenter(options)进行地图定位显示。options参数：

      • center(偏移坐标-必填)根据传入参数、
      • zoom(缩放层级-选填)、
      • duration(持续时间-选填)来以传入坐标为中心点偏移及层级缩放在持续时间内

      # 坐标定位示例

      <template>
    <div style="height: 400px; width: 100%">
        <!-- 地图组件 -->
        <OLMap :center="center" :zoomLevel="zoomLevel" ref="map">
            <!-- 基础底图高德瓦片 -->
            <AMapTileLayer />
            <!-- mark图层 -->
            <OLVectorLayer>
                <!-- 数据源 -->
                <OLVectorSource>
                    <!-- 点图形元素 -->
                    <OLFeature geoType="Point" :coordinates="coordinates" :properties="properties"/>
                </OLVectorSource>
            </OLVectorLayer>
            <!-- 工具栏 -->
            <OLBar position="top">
                <OLButton title="坐标查询1" icon="maki-fuel" @onClick="searchPos1"/>
                <OLButton title="坐标查询2" icon="maki-marker_stroked" @onClick="searchPos2"/>
            </OLBar>
        </OLMap>
    </div>
</template>

<script>
    export default {
        data: function() {
            return  {
                // 中心点坐标
                center: [116.302778, 40.050833],
                // 缩放层级
                zoomLevel: 13,
                // 是否显示mark标记
                searchShow: true,
                // mark 标记点属性经纬度坐标
                coordinates: [116.302778, 40.050833],
                // 点图形元素属性信息
                properties: {}
            }
        },

        methods: {
            searchPos1: function() {
                let coord = [116.303778, 40.051833];
                let options = {
                    center: coord,
                    zoom: 15
                }
                // 移动到指定的中心点
                this.$refs.map.moveCenter(options);

                let style = {
                    image: {
                        circle:{
                            radius: 10,
                            fill: 'pink',
                            stroke: {
                                color: 'blue',
                                width: 2
                            }
                        }
                    }
                }
                // 修改marker元素的坐标及样式
                this.coordinates = coord.slice();
                this.properties = {
                    style: style
                };
            },

            searchPos2: function() {
                let coord = [116.300778,40.049833];
                let options = {
                    center: coord,
                    zoom: 16
                }
                let style = {
                    image: {
                        icon:{
                            anchor: [0.5, 1],
                            src: '/images/560868.png',
                            crossOrigin: "anonymous"
                        }
                    }
                }
                this.$refs.map.moveCenter(options);
                this.searchShow = true;
                this.coordinates = coord;
                this.properties = {
                    style: style
                };
            }
        }
    }
</script>

      # 通过名称定位坐标示例

      <template>
    <div style="height: 400px; width: 100%">
    <!-- 地图组件 -->
      <OLMap :center="initCenter" ref="map">
        <!-- 基础底图高德瓦片 -->
        <AMapTileLayer />
        <!-- 矢量图层组件 -->
        <OLVectorLayer>
            <!-- 矢量数据源组件 -->
            <OLGeoJsonSource :features="features"></OLGeoJsonSource>
        </OLVectorLayer>
        <!-- mark层 -->
        <OLVectorLayer>
            <OLVectorSource>
                <!-- 点图形元素 -->
                <OLFeature v-if="searchShow" geoType="Point" :coordinates="markCoordinates" :properties="markerPops"/>
            </OLVectorSource>
        </OLVectorLayer>
        <OLBar position="top">
            <OLButton title="名称查询" icon="fa fa-map-marker" @onClick="searchName"/>
            <OLButton title="清除maker" icon="fa fa-trash" @onClick="searchShow=false"/>
        </OLBar>
      </OLMap>
    </div>
</template>

<script>
    export default {
        data: function() {
            return  {
                // 地图默认中心点
                initCenter: [116.302778, 40.050833],
                features: [{
                    "type": "Feature",
                    "properties": {
                        "id": "id_1",
                        "style": {
                            "image": {
                                "circle": {
                                    "radius": 5,
                                    "stroke": { "color": "blue", "width": 2 },
                                    "fill": 'red'
                                }
                            }
                        },
                        "name": "中恒博瑞",
                    },
                    "geometry": {
                        "type": "Point",
                        "coordinates": [116.302778,40.050833]
                    }
                }],
                // 是否显示mark标记
                searchShow: false,
                // mark 标记点经坐标
                markCoordinates: [],
                // mark点属性
                markerPops: {
                    style: {
                        image: {
                            icon:{
                                src: "/images/location_2.png",
                                anchor: [0.5, 1],
                            }
                        }
                    }
                }
            }
        },

        methods: {
            searchName(){
                let name = '中恒博瑞';
                // 筛选匹配名称Feature
                let matchFea = this.features.filter(fea => {
                    return fea.properties.name === name;
                })
                // 获取坐标
                let coord = matchFea[0].geometry.coordinates.slice();
                // 修改marker点的坐标
                this.markCoordinates = coord;
                // 展示marker点
                this.searchShow = true;
                // 移动地图到指定位置
                this.$refs.map.moveCenter({
                    center: coord,
                    zoom: 16
                });
            }
        }
    }
</script>

      # 事件说明

      # moveCenter(opt_options) 偏移中心方法

      opt_options 参数是一个对象，默认包含3个参数， 分别是center，zoom，duration
      center： 所偏移中心点坐标， —— 必填
      zoom： 偏移是地图缩放层级，默认为当前地图缩放层级 —— 选填
      duration： 持续时间， 默认为500 —— 选填

      # 备注

      # 设备搜索，若直接展示查找坐标 参照searchPos1() 写法， 若添加动画效果，参照searchPos2()写法
