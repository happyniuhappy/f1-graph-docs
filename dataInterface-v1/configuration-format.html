组态图数据格式基于jointJS框架的数据格式进行定义。

# 点图形

{
    //图元类型，必选
    "type": "electricity.DuanLuQi",
    //连接锚点
    "ports": {
        "items": [
            {
                //锚点分组
                "group": "terminal_1",
                //锚点位置参数，可选
                "args": {
                    "x": "50%",
                    "y": "0%"
                },
                //锚点id
                "id": "fc9a8827-28f7-42b3-b56a-a2f38caa4adc"
            },
            {
                "group": "terminal_2",
                "args": {
                    "x": "50%",
                    "y": "100%"
                },
                "id": "ac52823c-9458-4e8b-9ac1-a2de6ec669e1"
            }
        ]
    },
    // 位置坐标
    "position": {
        "x": 400,
        "y": 280
    },
    // 实例符号大小
    "size": {
        "width": 40,
        "height": 40
    },
    // 旋转角度
    "angle": 0,
    // 状态
    "markupState": 0,
    // 符号名称
    "symbolName": "断路器",
    //唯一标识
    "id": "d2947507-d8bb-425b-960f-5b89082855e8",
    //z-index层级
    "z": 3,
    //样式属性
    "attrs": {
        // 文字样式
        "text": {
            "ref-y": 0.5
        }
    }
}
