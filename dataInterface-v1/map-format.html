# WebGIS地图

采用符合OGC规范的GeoJSON格式作为数据交互的格式，可通过GeoJSON文件导入数据或从服务端获取GeoJSON数据，进行数据展示。

# GeoJSON格式

  • GeoJSON是一种对各种地理数据结构进行编码的格式，基于Javascript对象表示地理空间信息数据交换格式。

  • GeoJSON对象可以表示几何、特征或者特征集合。

  • GeoJSON支持几何类型：点、线、面、多点、多线、多面和几何集合。

  • GeoJSON里的特征包含一个几何对象和其他属性，特征集合表示一系列特征。

{
    "type": "FeatureCollection", // 固定写法
    "features": [
        {
            "geometry": {
                "coordinates": [
                    103.33815554369201,
                    35.939981843692
                ],
                "type": "Point"
            },
            "type": "Feature",
            "properties": {
                "textMinZoomLevel": 16,
                "devName": "110kV枣园变电站",
                "style": {
                    "image": {
                        "font-symbol": {
                            "radius": 10,

                "graphId": "241617",

                "textMaxZoomLevel": 20,
                "minZoomLevel": 12
            }
        }
    ]
}

# 图形平台

在GeoJSON基础之上，通过properties对象定义图形平台webgis地图中使用的属性信息。

# 样式定义

# Point

# circle图元符号
//circle符号渲染定义
{
    //固定格式，代表是一个图形元素
    "type": "Feature",
    //几何图形坐标定义
    "geometry": {
        //几何图形类型，包含Point、LineString、Polygon
        "type": "Point",
        //地理坐标，点是一个一纬数组
        "coordinates": [105.380859375, 31.57853542647338]
    },
    //图形元素的属性信息，除必要字段外，其他属性可根据实际需求自由组织
    "properties": {
        //图形元素唯一标识，不能重复
        "id": "ganTai_1",
        //图形显示到地图中的样式
        "style": {
            //点图形渲染图标
            "image": {
                //图元符号使用圆渲染
                "circle": {
                    //圆的边框颜色、边框线宽度
                    "stroke": { "color": "red", "width": 1 },
                    //圆的填充颜色
                    "fill": "#000000"
                }
            }
        }
    }
};

# 连接关系

连接关系是一条连接线，首尾连接点图形设备。

//定义连接线对象
var ganTaObj1 = {
    "type": "Feature",
    "geometry": {
        "type": 'Point',
        "coordinates": [105.6005859375, 30.65681556429287]
    },
    //属性内容可根据实际需求自由组织
    "properties": {
        id: "ganTaGraphId_1"
    }
};
//定义连接线对象
var ganTaObj2 = {
    "type": "Feature",
    "geometry": {
        "type": 'Point',
        "coordinates": [107.95166015624999, 31.98944183792288]
    },
    //属性内容可根据实际需求自由组织
    "properties": {
        id: "ganTaGraphId_2"
    }
};
//定义连接线对象
var lineObj = {
    "type": "Feature",
    "geometry": {
        "type": 'Point',
        "coordinates": [[105.6005859375, 30.65681556429287], [107.95166015624999, 31.98944183792288]]
    },
    //属性内容可根据实际需求自由组织
    "properties": {
        "id": "lineGraphId",
        //线的起点连接的对象，这里是定义的ganTaObj1对象的id
        "source": {
            //起点连接的Point图形Id
            "id": "ganTaiGraphId_1",
            //连接锚点Id，如果连接的Point不存在锚点，忽略此项
            "port": "portId@terminal_1"
        },
        //线的终点连接的对象，这里是定义的ganTaObj2对象id
        "target": {
            //起点连接的Point图形Id
            "id": "ganTaiGraphId_2",
            //连接锚点Id，如果连接的Point不存在锚点，忽略此项
            "port": "portId@terminal_2"
        }
    }
};

# 连接锚点

连接锚点是指电力图元中供连接线连接的热点。只存在于Point几何图形中。

var bianYaQiObj = {
    "type": "Feature",
    "geometry": {
        "type": "Point",
        "coordinates": [
            114.47813543273962,
            38.028923753497004
        ]
    },
    "properties": {
        "id": "C91AE327-85A0-0001-EED6-184A18ADC700",
        "style": {
            "image": {
                "svg-symbol": {
                    //
                    "symbolType": "electricity.ZhuShangBianYaQi(ZhuanYong)",
                    "stroke": {
                        "color": "blue",
                        "width": 1
                    }
                }
            }
        },
        //锚点定义
        "ports": {
            //锚点数组集合，有几个锚点就是几个数组对象
            "items": [
                {
                    //锚点分组名称定义
                    "group": "terminal_1",
                    //锚点参数
                    "args": {
                        //以图元左上角为原点，锚点x坐标与宽度的百分比
                        "x": "50%",
                        //以图元左上角为原点，锚点y坐标与高度的百分比
                        "y": "5.546566666666666%"
                    },
                    //锚点唯一标识
                    "id": "C91AE327-85A0-0001-EED6-184A18ADC700@terminal_1"
                },
                {
                    "group": "terminal_2",
                    "args": {
                        "x": "50%",
                        "y": "94.64093333333334%"
                    },
                    "id": "C91AE327-85A0-0001-EED6-184A18ADC700@terminal_2"
                }
            ]
        }
    }

# 文本样式

{
    "type": "Feature",
    "geometry": {
        "type": "Point",
        "coordinates": [105.380859375, 31.57853542647338]
    },
    //图形元素属性
    "properties": {
        //图形样式
        "style": {
            "text": {
                //文字放置，包含的值：point,line
                "placement": "point",
                //文字对齐，包含：left,right,center,endorstart
                "textAlign": "left",
                //文字位置基准线，包含：
                //bottom,top,middle,alphabetic,hanging,ideographic
                "textBaseline": "middle",
                //字体，包含字体样式风格、字体大小、字体名称
                "font": "normal12px宋体",
                //显示文字，缺省值使用name属性的值
                "text": "显示文字，缺省值使用name属性的值",
                //文字颜色
                "fill": "#aa3300",
                //文字描边颜色、描边边框宽度
                "stroke": { color: "#aa3300", width: 1 },
                //经度方向上的偏移量，单位像素
                "offsetX": 0,
                //纬度方向上的偏移量，单位像素
                "offsetY": 0,
                //文字标签背景颜色，缺省无背景颜色
                "backgroundFill": null,
                //文字标签背景边框颜色、宽度，缺省无背景颜色
                "backgroundStroke": { color: "#aa3300", width: 1 },
                //文字缩放比例
                "scale": 1,
                //是否随地图视图旋转
                "rotateWithView": false,
                //文字旋转角度，弧度制
                "rotation": 0,
                //距离文字内容的缝隙值，
                //数组中值的顺序为[top,right,bottom,left]
                "padding": [0, 0, 0, 0],
                //对于多边形标签或当placement设置为“line”时，
                //允许文本超出标签位置处多边形的宽度或所沿路径的长度。
                "overflow": false
            }
        }
    }
};
